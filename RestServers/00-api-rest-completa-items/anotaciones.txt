INSTALAR TYPESCRIPT DE MANERA GLOBAL    
    npm install typescript -g
INSTALAR TS-NODE DE MANERA GLOBAL
    npm install ts-node -g
INSTALAR NODEMON DE MANERA GLOBAL
    npm install nodemon -g
INICIALIZAMOS PROYECTO
    tsc --init
    npm init -yes
DEPENDENCIAS
    npm install express cors dotenv multer mongoose --save
DEV DEPENDENCIES
    npm install @types/express @types/cors @types/dotenv @types/multer @types/multer @types/mongoose -D

GENERAR BUILD DE PRODUCCION
npm run build
